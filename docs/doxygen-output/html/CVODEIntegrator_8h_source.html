327 std::string decodeSundialsError(SundialsType* solver, int cvodeError, bool exInfo) {
328 std::string result;
329 std::stringstream ss;
330 ss << (int) solver->getValue("maximum_num_steps");
331 std::string max_steps = ss.str();
332
333 switch (cvodeError) {
334 case CV_TOO_MUCH_WORK:
335 result = "CV_TOO_MUCH_WORK";
336 if (exInfo) {
337 result += ": The solver took mxstep (" + max_steps + ") internal steps but " +
338 "could not reach tout.";
339 }
340 break;
341 case CV_TOO_MUCH_ACC:
342 result = "CV_TOO_MUCH_ACC";
343 if (exInfo) {
344 result += ": The solver could not satisfy the accuracy "
345 "demanded by the user for some internal step.";
346 }
347 break;
348 case CV_ERR_FAILURE:
349 result = "CV_ERR_FAILURE";
350 if (exInfo) {
351 result += ": Error test failures occurred too many times "
352 "(= MXNEF = 7) during one internal time step or"
353 "occurred with |h| = hmin.";
354 }
355 break;
356 case CV_CONV_FAILURE:
357 result = "CV_CONV_FAILURE";
358 if (exInfo) {
359 result += ": Convergence test failures occurred too many "
360 "times (= MXNCF = 10) during one internal time"
361 "step or occurred with |h| = hmin.";
362 }
363 break;
364 case CV_LINIT_FAIL:
365 result = "CV_LINIT_FAIL";
366 if (exInfo) {
367 result += ": The linear solver's initialization function "
368 "failed.";
369 }
370 break;
371 case CV_LSETUP_FAIL:
372 result = "CV_LSETUP_FAIL";
373 if (exInfo) {
374 result += ": The linear solver's setup routine failed in an "
375 "unrecoverable manner.";
376 }
377 break;
378 case CV_LSOLVE_FAIL:
379 result = "CV_LSOLVE_FAIL";
380 if (exInfo) {
381 result += ": The linear solver's solve routine failed in an "
382 "unrecoverable manner.";
383 }
384 break;
385 case CV_RHSFUNC_FAIL:
386 result = "CV_RHSFUNC_FAIL";
387 break;
388 case CV_FIRST_RHSFUNC_ERR:
389 result = "CV_FIRST_RHSFUNC_ERR";
390 break;
391 case CV_REPTD_RHSFUNC_ERR:
392 result = "CV_REPTD_RHSFUNC_ERR";
393 break;
394 case CV_UNREC_RHSFUNC_ERR:
395 result = "CV_UNREC_RHSFUNC_ERR";
396 break;
397 case CV_RTFUNC_FAIL:
398 result = "CV_RTFUNC_FAIL";
399 break;
400 case CV_MEM_FAIL:
401 result = "CV_MEM_FAIL";
402 break;
403 case CV_MEM_NULL:
404 result = "CV_MEM_NULL";
405 if (exInfo) {
406 result += ": The cvode_mem argument was NULL.";
407 }
408 break;
409 case CV_ILL_INPUT:
410 result = "CV_ILL_INPUT";
411 if (exInfo) {
412 result += ": One of the inputs to CVode is illegal. This "
413 "includes the situation when a component of the "
414 "error weight vectors becomes < 0 during "
415 "internal time-stepping. It also includes the "
416 "situation where a root of one of the root "
417 "functions was found both at t0 and very near t0. "
418 "The ILL_INPUT flag will also be returned if the "
419 "linear solver routine CV--- (called by the user "
420 "after calling CVodeCreate) failed to set one of "
421 "the linear solver-related fields in cvode_mem or "
422 "if the linear solver's init routine failed. In "
423 "any case, the user should see the printed "
424 "error message for more details.";
425 }
426 break;
427 case CV_NO_MALLOC:
428 result = "CV_NO_MALLOC";
429 if (exInfo) {
430 result += ": indicating that cvode_mem has not been "
431 "allocated (i.e., CVodeInit has not been "
432 "called).";
433 }
434 break;
435 case CV_BAD_K:
436 result = "CV_BAD_K";
437 if (exInfo) {
438 result += ": k is not in the range 0, 1, ..., qu.";
439 }
440 break;
441 case CV_BAD_T:
442 result = "CV_BAD_T";
443 if (exInfo) {
444 result += ": t is not in the interval [tn-hu,tn].";
445 }
446 break;
447 case CV_BAD_DKY:
448 result = "CV_BAD_DKY";
449 if (exInfo) {
450 result += ": The dky argument was NULL.";
451 }
452 break;
453 case CV_TOO_CLOSE:
454 result = "CV_TOO_CLOSE:";
455 break;
456 default:
457 result = "UNKNOWN_CODE";
458 break;
459 }
460 return result;
461 }
