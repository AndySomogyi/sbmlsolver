118  virtual double integrate(double t0, double h) {
119  int internal_steps = getValueAsInt("subdivision_steps");
120  if (model == (rr::ExecutableModel*)NULL) return 0;
121 
122  double finalTimeEnd;
123 
124  h /= internal_steps;
125  for (int subdiv = 0; subdiv < internal_steps; ++subdiv) {
126  // evaluate and copy the rate of change of the state vector
127  // rate into the local buffer. If the 2nd argument is NULL,
128  // the current model state is used to evaluate the
129  // state vector rate.
130  model->getStateVectorRate(t0, NULL, rateBuffer);
131 
132  // copy the current state vector into a local buffer
133  model->getStateVector(stateBufferBegin);
134 
135  // perform the Euler integration step, i.e.
136  // y_{n+1} = y_{n} + h * y'_{n}
137  for (int i = 0; i < stateVectorSize; ++i) {
138  stateBufferEnd[i] = stateBufferBegin[i] + h * rateBuffer[i];
139  }
140 
141  // set the model state to the newly calculated state
142  model->setStateVector(stateBufferEnd);
143 
144  // update the model time to the new time
145  double timeEnd = t0 + h;
146  model->setTime(timeEnd);
147 
148  // if we have a client, notify them that we have taken
149  // a time step
150  if (listener) {
151  listener->onTimeStep(this, model, timeEnd);
152  }
153 
154  // events
155  bool triggered = false;
156 
157  model->getEventTriggers(eventStatus.size(), NULL, eventStatus.size() ? &eventStatus[0] : NULL);
158  for (int k_ = 0; k_ < eventStatus.size(); ++k_) {
159  if (eventStatus.at(k_)) {
160  triggered = true;
161  //std::cerr << "Triggered" << std::endl;
162  }
163  }
164 
165  if (triggered) {
166  // applyEvents takes the list of events which were previously triggered
167  //std::cerr << "An event was triggered at " << t0 << std::endl;
168  applyEvents(timeEnd, previousEventStatus);
169  }
170 
171  if (eventStatus.size()) {
172  previousEventStatus = eventStatus;
173  }
174 
175  finalTimeEnd = timeEnd;
176  }
177 
178  return finalTimeEnd;
179  }
