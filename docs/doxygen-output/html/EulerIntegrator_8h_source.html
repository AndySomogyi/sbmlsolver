124 virtual double integrate(double t0, double h) override {
125 int internal_steps = getValue("subdivision_steps");
126 if (mModel == (rr::ExecutableModel *) NULL) return 0;
127
128 double finalTimeEnd;
129
130 h /= internal_steps;
131 for (int subdiv = 0; subdiv < internal_steps; ++subdiv) {
132 // evaluate and copy the rate of change of the state std::vector
133 // rate into the local buffer. If the 2nd argument is NULL,
134 // the current mModel state is used to evaluate the
135 // state std::vector rate.
136 mModel->getStateVectorRate(t0, NULL, rateBuffer);
137
138 // copy the current state std::vector into a local buffer
139 mModel->getStateVector(stateBufferBegin);
140
141 // perform the Euler integration step, i.e.
142 // y_{n+1} = y_{n} + h * y'_{n}
143 for (int i = 0; i < stateVectorSize; ++i) {
144 stateBufferEnd[i] = stateBufferBegin[i] + h * rateBuffer[i];
145 }
146
147 // set the mModel state to the newly calculated state
148 mModel->setStateVector(stateBufferEnd);
149
150 // update the mModel time to the new time
151 double timeEnd = t0 + h;
152 mModel->setTime(timeEnd);
153
154 // if we have a client, notify them that we have taken
155 // a time step
156 if (listener) {
157 listener->onTimeStep(this, mModel, timeEnd);
158 }
159
160 // events
161 bool triggered = false;
162
163 mModel->getEventTriggers(eventStatus.size(), NULL, eventStatus.size() ? &eventStatus[0] : NULL);
164 for (int k_ = 0; k_ < eventStatus.size(); ++k_) {
165 if (eventStatus.at(k_)) {
166 triggered = true;
167 //std::cerr << "Triggered" << std::endl;
168 }
169 }
170
171 if (triggered) {
172 // applyEvents takes the list of events which were previously triggered
173 //std::cerr << "An event was triggered at " << t0 << std::endl;
174 applyEvents(timeEnd, previousEventStatus);
175 }
176
177 if (!eventStatus.empty()) {
178 previousEventStatus = eventStatus;
179 }
180
181 finalTimeEnd = timeEnd;
182 }
183
184 return finalTimeEnd;
185 }
