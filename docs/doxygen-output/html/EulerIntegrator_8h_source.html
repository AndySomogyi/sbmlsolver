119  virtual double integrate(double t0, double h) {
120  int internal_steps = getValue("subdivision_steps");
121  if (mModel == (rr::ExecutableModel*)NULL) return 0;
122 
123  double finalTimeEnd;
124 
125  h /= internal_steps;
126  for (int subdiv = 0; subdiv < internal_steps; ++subdiv) {
127  // evaluate and copy the rate of change of the state std::vector
128  // rate into the local buffer. If the 2nd argument is NULL,
129  // the current mModel state is used to evaluate the
130  // state std::vector rate.
131  mModel->getStateVectorRate(t0, NULL, rateBuffer);
132 
133  // copy the current state std::vector into a local buffer
134  mModel->getStateVector(stateBufferBegin);
135 
136  // perform the Euler integration step, i.e.
137  // y_{n+1} = y_{n} + h * y'_{n}
138  for (int i = 0; i < stateVectorSize; ++i) {
139  stateBufferEnd[i] = stateBufferBegin[i] + h * rateBuffer[i];
140  }
141 
142  // set the mModel state to the newly calculated state
143  mModel->setStateVector(stateBufferEnd);
144 
145  // update the mModel time to the new time
146  double timeEnd = t0 + h;
147  mModel->setTime(timeEnd);
148 
149  // if we have a client, notify them that we have taken
150  // a time step
151  if (listener) {
152  listener->onTimeStep(this, mModel, timeEnd);
153  }
154 
155  // events
156  bool triggered = false;
157 
158  mModel->getEventTriggers(eventStatus.size(), NULL, eventStatus.size() ? &eventStatus[0] : NULL);
159  for (int k_ = 0; k_ < eventStatus.size(); ++k_) {
160  if (eventStatus.at(k_)) {
161  triggered = true;
162  //std::cerr << "Triggered" << std::endl;
163  }
164  }
165 
166  if (triggered) {
167  // applyEvents takes the list of events which were previously triggered
168  //std::cerr << "An event was triggered at " << t0 << std::endl;
169  applyEvents(timeEnd, previousEventStatus);
170  }
171 
172  if (!eventStatus.empty()) {
173  previousEventStatus = eventStatus;
174  }
175 
176  finalTimeEnd = timeEnd;
177  }
178 
179  return finalTimeEnd;
180  }
