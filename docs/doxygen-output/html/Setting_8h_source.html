212 template<class As>
213 As getAs() const{
214 const std::type_info &inf = typeInfo();
215 return visit([&](auto &&val) {
216 if constexpr (std::is_convertible_v<decltype(val), As>) {
217 // if we have an int
218 if (auto intValue = std::get_if<int>(&value_)) {
219 // and its negative,
220 if (*intValue < 0) {
221 // we cannot convert to unsigned int or unsigned long
222 if (typeid(As) == typeid(unsigned int) || typeid(As) == typeid(unsigned long)) {
223 throw std::bad_variant_access{};
224 return As{};
225 }
226 }
227 }
228
229 // if we have a long,
230 if (auto lValue = std::get_if<std::int64_t>(&value_)) {
231 // and its negative, we cannot convert to unsigned int or unsigned long
232 if (*lValue < 0) {
233 if (typeid(As) == typeid(unsigned int) || typeid(As) == typeid(unsigned long)) {
234 throw std::bad_variant_access{};
235 return As{}; // must be present
236 }
237 }
238 // furthermore, if we have a long which has a value greater than
239 // that of int32 maximum, we have a problem and throw
240 if (*lValue > ((std::int64_t) (std::numeric_limits<int>::max)())) {
241 throw std::bad_variant_access{}; // has annoying private constructor so we can't add arguments
242 return As{}; // must be present
243 }
244 }
245 // if we have a double, with value greater than std::numeric_limits<float>::max
246 // and we try to convert to float, we have an error
247 if (auto lValue = std::get_if<float>(&value_)) {
248 if (*lValue > ((std::numeric_limits<float>::max)())) {
249 throw std::bad_variant_access{};
250 return As{}; // must be present
251 }
252 }
253 return As(val);
254 } else {
255 std::ostringstream os;
256 os << "Setting::getAs:TypeError. You have requested the conversion "
257 "of a \"" << typeid(decltype(val)).name() << "\" to a ";
258 os << "\"" << typeid(As).name() << "\" but this Setting contains ";
259 os << "a \"" << inf.name() << "\". Note, see Setting::toString() "
260 "for string representation." << std::endl;
261 // would prefer to throw std::bad_variant_access, but it seems
262 // it does not have the appropriate constructor (?)
263 throw std::invalid_argument(os.str());
264 return As{}; // oddly enough, this *is* necessary
265 }
266 });
267 }
