97 template <size_t N>
98 llvm::BasicBlock *codeGenHeader(const char* functionName,
99 llvm::Type *retType,
100 llvm::Type* (&argTypes)[N],
101 const char* (&argNames)[N],
102 llvm::Value* (&args)[N])
103 {
104 // make the set init value function
105
106 llvm::FunctionType *funcType = llvm::FunctionType::get(retType, argTypes, false);
107 function = llvm::Function::Create(funcType,
108 llvm::Function::ExternalLinkage,
109 functionName, module);
110
111 // Create a new basic block to start insertion into.
112 llvm::BasicBlock *basicBlock = llvm::BasicBlock::Create(context,"entry", function);
113 builder.SetInsertPoint(basicBlock);
114
115 assert(function->arg_size() == N);
116
117 int i = 0;
118 for (llvm::Function::arg_iterator ai = function->arg_begin();
119 ai != function->arg_end(); ++ai, ++i)
120 {
121 llvm::Value *arg = ai;
122 arg->setName(argNames[i]);
123 args[i] = arg;
124 }
125
126 return basicBlock;
127 }
128
133 llvm::BasicBlock *codeGenVoidModelDataHeader(const char* functionName,llvm::Value* &modelData)
134 {
135 llvm::Type *argTypes[] = {
136 llvm::PointerType::get(
137 ModelDataIRBuilder::getStructType(module), 0)
138 };
139
140 const char *argNames[] = { "modelData" };
141
142 llvm::Value *args[] = { 0 };
143
144 llvm::BasicBlock *basicBlock = codeGenHeader(functionName,
145 llvm::Type::getVoidTy(context),
146 argTypes, argNames, args);
147
148 modelData = args[0];
149 return basicBlock;
150 }
