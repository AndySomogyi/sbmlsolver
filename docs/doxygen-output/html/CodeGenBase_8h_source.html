115  llvm::Type *retType,

117  const char* (&argNames)[N],

123  function = llvm::Function::Create(funcType,

125  functionName, module);
126 
127  // Create a new basic block to start insertion into.
128  llvm::BasicBlock *basicBlock = llvm::BasicBlock::Create(context,"entry", function);
129  builder.SetInsertPoint(basicBlock);

131  assert(function->arg_size() == N);
132 
133  int i = 0;

135  ai != function->arg_end(); ++ai, ++i)
136  {
137  llvm::Value *arg = ai;
138  arg->setName(argNames[i]);
139  args[i] = arg;
140  }
