103  template<class KinsolSteadyStateSolverType>
104  static int kinsolDyDtFcn(N_Vector stateVecIn, N_Vector stateVecOut, void *userData) {
105  // we need to use the state std::vector y (chosen by the solver)
106  // to compute the rates dydt and then update the state std::vector
107  // using the rates.
108  double *y = NV_DATA_S(stateVecIn);
109 
110  // initialize the out NVector with numerical limits for double
111  N_VConst(std::numeric_limits<double>::max(), stateVecOut);
112  double *dydt = NV_DATA_S(stateVecOut);
113 
114  // cast user data back into our solver type
115  auto solver = (KinsolSteadyStateSolverType *) userData;
116 
117  assert(solver && "userData pointer is nullptr in callback kinsolDyDtFcn");
118 
119  ExecutableModel *model = solver->mModel;
120 
121  assert(model && "model is nullptr");
122 
123  // collect the number of states in this model
124  int numStates = stateVecIn->ops->nvgetlength(stateVecIn);
125 
126  model->getStateVectorRate(model->getTime(), y, dydt);
127 
128  for (int i = 0; i < numStates; i++) {
129  if (dydt[i] == std::numeric_limits<double>::max()) {
130  std::ostringstream err;
131  err << __FILE__":" << __LINE__ << ":" << "kinsolDyDtFcn";
132  err << ": steady state solver \"" << solver->getName()
133  << "\". The rate for variable \"" << model->getStateVectorId(i)
134  << "\" is equal to the numerical ";
135  err << "limits for a double." << std::endl;
136  throw KinsolException(err.str());
137  }
138  }
139  rrLog(Logger::LOG_TRACE) << __FUNCTION__ << ", model: " << model;
140 
141  return KIN_SUCCESS;
142  };
