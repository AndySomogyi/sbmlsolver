41  void insert(IndexType idx, Matrix <DataType> mat) {
42  // enforce unique indexes by override existing idx
43  auto it = std::find(index_.begin(), index_.end(), idx);
44  if (it != index_.end()) {
45  auto pos = std::distance(index_.begin(), it);
46  data_[pos] = mat;
47  return;
48  }
49  if (!index_.empty()) {
50  // All sizes are compared to index 0, so when initializing
51  // from empty, we do not check integrity of user input
52  // need to check that all Z have dimensions x and y
53  if (mat.numRows() != numRows()) {
54  std::ostringstream err;
55  err << "Cannot insert a matrix with " << mat.numRows();
56  err << " rows into a Matrix3D that has " << numRows() << "rows";
57  throw std::invalid_argument(err.str());
58  }
59  if (mat.numCols() != numCols()) {
60  std::ostringstream err;
61  err << "Cannot insert a matrix with " << mat.numCols();
62  err << " rows into a Matrix3D that has " << numCols() << "rows";
63  throw std::invalid_argument(err.str());
64  }
65  }
66  index_.push_back(idx);
67  data_.push_back(mat);
68  }
