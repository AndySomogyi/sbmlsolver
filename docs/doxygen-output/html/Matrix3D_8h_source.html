40 void insert(IndexType idx, Matrix <DataType> mat) {
41 // enforce unique indexes by override existing idx
42 auto it = std::find(index_.begin(), index_.end(), idx);
43 if (it != index_.end()) {
44 auto pos = std::distance(index_.begin(), it);
45 data_[pos] = mat;
46 return;
47 }
48 if (!index_.empty()) {
49 // All sizes are compared to index 0, so when initializing
50 // from empty, we do not check integrity of user input
51 // need to check that all Z have dimensions x and y
52 if (mat.numRows() != numRows()) {
53 std::ostringstream err;
54 err << "Cannot insert a matrix with " << mat.numRows();
55 err << " rows into a Matrix3D that has " << numRows() << "rows";
56 throw std::invalid_argument(err.str());
57 }
58 if (mat.numCols() != numCols()) {
59 std::ostringstream err;
60 err << "Cannot insert a matrix with " << mat.numCols();
61 err << " rows into a Matrix3D that has " << numCols() << "rows";
62 throw std::invalid_argument(err.str());
63 }
64 }
65 index_.push_back(idx);
66 data_.push_back(mat);
67 }

184 Matrix <DataType> &getItem(IndexType idx) {
185 // first check if idx in index
186 if (std::find(index_.begin(), index_.end(), idx) == index_.end()) {
187 std::ostringstream err;
188 err << "Index \"" << idx << "\" has been requested but is not present in this Matrix3D.";
189 throw std::invalid_argument(err.str());
190 }
191 // get index of idx
192 int pos = std::distance(index_.begin(), std::find(index_.begin(), index_.end(), idx));

283 bool operator==(Matrix3D<IndexType, DataType> &other) {
284 if (numRows() != other.numRows() || numCols() != other.numCols() || numZ() != other.numZ()) {
285 return false;

311 bool almostEquals(Matrix3D<double, double> &other, double tol) {
312 if (numRows() != other.numRows() || numCols() != other.numCols() || numZ() != other.numZ()) {
313 return false;
