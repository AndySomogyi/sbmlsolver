67 template <typename Derived, typename FunctionPtrType>
68 llvm::Value *GetEventValueCodeGenBase<Derived, FunctionPtrType>::codeGen()
69 {
70  // make the set init value function
71  llvm::Type *argTypes[] = {
72  llvm::PointerType::get(ModelDataIRBuilder::getStructType(this->module), 0),
73  llvm::Type::getInt32Ty(this->context)
74  };
75 
76  const char *argNames[] = {
77  "modelData", Derived::IndexArgName
78  };
79 
80  llvm::Value *args[] = {0, 0};
81 
82  llvm::Type *retType = static_cast<Derived*>(this)->getRetType();
83 
84  llvm::BasicBlock *entry = this->codeGenHeader(Derived::FunctionName, retType,
85  argTypes, argNames, args);
86 
87  const libsbml::ListOfEvents *events = this->model->getListOfEvents();
88 
89  ModelDataLoadSymbolResolver resolver(args[0], this->modelGenContext);
90 
91  ASTNodeCodeGen astCodeGen(this->builder, resolver, this->modelGenContext, args[0]);
92 
93  // default, return NaN
94  llvm::BasicBlock *def = llvm::BasicBlock::Create(this->context, "default", this->function);
95  this->builder.SetInsertPoint(def);
96 
97  llvm::Value *defRet = static_cast<Derived*>(this)->createRet(0);
98  this->builder.CreateRet(defRet);
99 
100  // write the switch at the func entry point, the switch is also the
101  // entry block terminator
102  this->builder.SetInsertPoint(entry);
103 
104  llvm::SwitchInst *s = this->builder.CreateSwitch(args[1], def, events->size());
105 
106  for (uint i = 0; i < events->size(); ++i)
107  {
108  char block_name[64];
109  std::sprintf(block_name, "event_%i_block", i);
110  llvm::BasicBlock *block = llvm::BasicBlock::Create(this->context, block_name, this->function);
111  this->builder.SetInsertPoint(block);
112  resolver.flushCache();
113 
114  const libsbml::Event *event = events->get(i);
115 
116  llvm::Value* value = static_cast<Derived*>(this)->getMath(event, astCodeGen);
117 
118  // convert type to return type
119  value = static_cast<Derived*>(this)->createRet(value);
120 
121  this->builder.CreateRet(value);
122  s->addCase(llvm::ConstantInt::get(llvm::Type::getInt32Ty(this->context), i), block);
123  }
124 
125  return this->verifyFunction();
126 }
