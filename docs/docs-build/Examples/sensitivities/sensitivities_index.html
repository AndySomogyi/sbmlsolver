  1############################################
  2# Time series sensitivities example
  3#
  4
  5"""
  6In this example we show you how to use roadrunners time series sensitivities
  7feature. Roadrunner will integrate the model and compute sensitivities at
  8each time step. You can select which parameters you want sensitivities for
  9and you can have the return matrix (a 3D numpy.ndarray) as the kth order
 10derivative of the sensitivity matrix.
 11"""
 12
 13import roadrunner
 14from roadrunner import RoadRunner
 15from roadrunner.testing.TestModelFactory import TestModelFactory, getAvailableTestModels
 16
 17# For this demonstration we use a toy sbml model prepackaged with roadrunner
 18
 19# note: use getAvailableTestModels() to see options for test models.
 20sbml = TestModelFactory("OpenLinearFlux").str() # get the test model's sbml string
 21
 22# print(sbml) # uncomment if you want to see the sbml
 23
 24# Create a RoadRunner instance
 25model = RoadRunner(sbml)
 26sensSolver = model.getSensitivitySolver()
 27
 28# start integration with sensitivities at 0, end at 10 and collect 11 data points
 29# The return type is a 4-Tuple, a time vector and a 3D sensitivity matrix with
 30# dimensions time x parameters x variables - that is, an array of matrices corresponding
 31# to the time points in the time vector with parameters down the rows and model variables (species)
 32# accross the columns. The third and forth elements are row and column names respectively.
 33time, sens, rownames, colnames = model.timeSeriesSensitivities(0, 10, 11)
 34
 35print("time: ", time)
 36print("rownames: ", rownames)
 37print("colnames: ", colnames)
 38
 39# Note that in this model, the parameter kb is not used in any
 40# rate law and therefore the sensitivities of model variables
 41# to kb is always 0
 42print(sens)
 43
 44# By default, all model parameters are automatically selected.
 45# You can be more selective about which parameters to compute sensitivities for.
 46# This will have a performance advantage, because sensitivities are only computed
 47# for the parameters specified here.
 48time, sens, rownames, colnames = model.timeSeriesSensitivities(0, 10, 11, params=["kin"])
 49
 50print("time: ", time)
 51print("rownames: ", rownames)
 52print("colnames: ", colnames)
 53print(sens)
 54
 55# The return type, by default, returns the k=0th derivative of the
 56# sensitivity matrix at each time point. The kth order derivative
 57# be obtained instead by using the k parameter
 58time, sens, rownames, colnames = model.timeSeriesSensitivities(0, 10, 11, k=1)
 59
 60print("time: ", time)
 61print("rownames: ", rownames)
 62print("colnames: ", colnames)
 63print(sens)
 64
 65# RoadRunner time series sensitivities uses cvodes under the hood. By default,
 66# we use the "simultaneous" method, whereby sensitivity equations are solved at the same
 67# time as the variables in the ODE's. This can be changed so that the variables
 68# are solved first followed by the sensitivity equations; aka in a staggered approach.
 69sensSolver.setValue("sensitivity_method", "staggered")
 70time, sens, rownames, colnames = model.timeSeriesSensitivities(0, 10, 11, k=0)
 71
 72print("using a staggered approach")
 73print("time: ", time)
 74print("rownames: ", rownames)
 75print("colnames: ", colnames)
 76print(sens)
 77sensSolver.setValue("sensitivity_method", "simultaneous") # return to simultaneous mode
 78
 79
 80# RoadRunner via cvodes uses finite differences for approximating gradients.
 81# By default we use the "centered" method, while the alternative is the "forward" method
 82# Both methods are implemented in cvodes and automatically figure out the best value of delta
 83# (perturbation amount) to use. Moreover, there is an adaptive switching mechanism in use
 84# that makes it very unlikely that users should need to modify this setting.
 85sensSolver.setValue("DQ_method", "forward")
 86time, sens, rownames, colnames = model.timeSeriesSensitivities(0, 10, 11, k=0)
 87
 88
 89print("using forward method of finite different approximation")
 90print("time: ", time)
 91print("rownames: ", rownames)
 92print("colnames: ", colnames)
 93print(sens)
 94sensSolver.setValue("DQ_method", "centered") # put back the default value
 95
 96# By default, RoadRunner uses the sundials "newton" nonlinear solver.
 97# This can be changed to the fixed_point nonlinear solver
 98sensSolver.setValue("nonlinear_solver", "fixed_point")
 99time, sens, rownames, colnames = model.timeSeriesSensitivities(0, 10, 11, k=0)
100
101print("using the fixed point nonlinear solver")
102print("time: ", time)
103print("rownames: ", rownames)
104print("colnames: ", colnames)
105print(sens)
