86 inline unsigned long MTRand_int32::twiddle(unsigned long u, unsigned long v) const {
87  return (((u & 0x80000000UL) | (v & 0x7FFFFFFFUL)) >> 1)
88  ^ ((v & 1UL) ? 0x9908B0DFUL : 0x0UL);
89 }

91 inline unsigned long MTRand_int32::rand_int32() const { // generate 32 bit random int
92  if (p == n) gen_state(); // new state vector needed
93 // gen_state() is split off to be non-inline, because it is only called once
94 // in every 624 calls and otherwise irand() would become too big to get inlined
95  unsigned long x = state[p++];
96  x ^= (x >> 11);
97  x ^= (x << 7) & 0x9D2C5680UL;
98  x ^= (x << 15) & 0xEFC60000UL;
99  return x ^ (x >> 18);
100 }
